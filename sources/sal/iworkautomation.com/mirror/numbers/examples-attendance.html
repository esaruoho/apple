Attendance Table for Contacts Group

This script is designed for teachers, or others, who require the keeping of an attendance record for a group of people, listed in the user’s Contacts application.

It provides examples of these techniques:

  • Construction of a new table (95-96)
  • The import of data from another application (19-59)
  • The use of assigned data formats for cells (132)
  • The use of contextual formulas (144)
  • The use of the sort command (150)

Here’s the script:

Create an Attendance Table for a Chosen Contact Group
  
01try
02 tell application id "com.apple.iWork.Numbers"
03 activate
04 
05 if not (exists document 1) then error number 1000
06 
07 display dialog "This script will create a new table for tracking " & ¬
08 "the attendance of the people in a chosen Contacts group." & ¬
09 return & return & ¬
10 "Should the table include Saturdays and Sundays?" buttons ¬
11 {"Cancel", "Omit Weekends", "Include Weekends"} default button 3
12 if the button returned of the result is "Include Weekends" then
13 set includeWeekends to true
14 else
15 set includeWeekends to false
16 end if
17 end tell
18 
19 -- launch Contacts if not already open
20 if application id "com.apple.AddressBook" is not running then
21 set AddressBookWasOpen to false
22 tell application id "com.apple.AddressBook" to launch
23 else
24 set AddressBookWasOpen to true
25 end if
26 
27 -- get the name of every Contacts group
28 tell application id "com.apple.AddressBook"
29 set the AddressBookGroupNames to the name of every group
30 end tell
31 if the AddressBookGroupNames is {} then error number 1001
32 
33 -- prompt user to pick group
34 tell application id "com.apple.iWork.Numbers"
35 activate
36 set the chosenGroup to ¬
37 (choose from list the AddressBookGroupNames ¬
38 with prompt "Pick the Contacts group to be used for the table:")
39 if the chosenGroup is false then error number -128
40 set the chosenGroup to the chosenGroup as string
41 end tell
42 
43 -- extract the full names of the group’s people
44 tell application id "com.apple.AddressBook"
45 set thesePeople to every person of group chosenGroup
46 if thesePeople is {} then error number 1002
47 set the fullNames to {}
48 repeat with i from 1 to the count of thesePeople
49 set thisPerson to item i of thesePeople
50 set the end of the fullNames to ¬
51 (last name of thisPerson) & ", " & (first name of thisPerson)
52 end repeat
53 set the peopleCount to the count of the fullNames
54 end tell
55 
56 -- close Contacts if it wasn’t previously open
57 if AddressBookWasOpen is false then
58 tell application id "com.apple.AddressBook" to quit
59 end if
62 
63 -- generate the column titles
64 set tempDate to the current date
65 set the columnTitles to {}
66 set day of tempDate to 1
67 set thisMonth to the month of tempDate
68 repeat until month of tempDate is not thisMonth
69 set thisDay to day of tempDate
70 set thisWeekday to the (weekday of tempDate) as string
71 if includeWeekends is true or (includeWeekends is false and ¬
72 thisWeekday is not in {"Saturday", "Sunday"}) then
73 set the dayTitle to ¬
74 (thisDay & "-" & (text 1 thru 1 of thisWeekday)) as string
75 set the end of the columnTitles to the dayTitle
76 end if
77 set tempDate to tempDate + (1 * days)
78 end repeat
79 
80 set the dayCount to the count of the columnTitles
81 
82 set the tableName to chosenGroup & space & ¬
83 (text 1 thru 3 of (thisMonth as string) & ¬
84 "-" & year of (the current date)) as string
85 set tableName to my incrementtableNameIfNeeded(tableName, "-")
86 
87 tell application id "com.apple.iWork.Numbers"
88 activate
89 
90 tell document 1
91 tell the active sheet
92 
93 -- make and populate the table
94 -- include extra column for total of missed days
95 set thisTable to make new table with properties ¬
96 {column count:dayCount + 2, row count:peopleCount + 1, name:tableName}
97 
98 tell thisTable
99 -- set any global cell properties
100 set the height of every row to 24
101 set the vertical alignment of every row to center
102 set the alignment of every row to center
103 -- insert data
104 tell row 1
105 -- set specific properties for the title row
106 set the vertical alignment to center
107 set the alignment to center
108 -- insert column titles
109 repeat with i from 2 to dayCount + 1
110 set value of cell i to item (i - 1) of the columnTitles
111 end repeat
112 end tell
113 tell column 1
114 -- set specific properties for the title column
115 set the vertical alignment to center
116 set the alignment to left
117 set the width to 128
118 -- insert full names
119 repeat with i from 2 to peopleCount + 1
120 set value of cell i to item (i - 1) of the fullNames
121 end repeat
122 end tell
123 tell columns 2 thru -2
124 set width to 36
125 end tell
126 
127 -- get the range of the cells to set
128 set the rangeStart to the name of cell 2 of column 2
129 set the rangeEnd to the name of last cell of column -2
130 -- set format of cells
131 tell range (rangeStart & ":" & rangeEnd)
132 set format to checkbox
133 end tell
134 
135 -- Set the style and formula of the total missed days column
136 tell last column
137 set the width to 60
138 set the value of cell 1 to "MISSED"
139 set alignment to center
140 end tell
141 repeat with i from 2 to the count of rows
142 tell row i
143 set the rangeString to (name of cell 2 & ":" & name of cell -2)
144 set the formulaString to "=COUNTIF(" & rangeString & ", FALSE)"
145 set the value of the last cell to formulaString
146 end tell
147 end repeat
148 
149 -- sort the table based on student names
150 sort by column 1 direction ascending
151 
152 end tell
153 end tell
154 end tell
155 return thisTable
156 end tell
157on error errorMessage number errorNumber
158 if errorNumber is 1000 then
159 set alertString to "MISSING NUMBERS RESOURCE"
160 set errorMessage to "Please create or open a document before running this script."
161 else if errorNumber is 1001 then
162 set alertString to "MISSING CONTACTS RESOURCE"
163 set errorMessage to "There are no groups in your Contacts."
164 else if errorNumber is 1002 then
165 set alertString to "MISSING CONTACTS RESOURCE"
166 set errorMessage to "There are no people in the chosen Contacts group."
167 else
168 set alertString to "EXECUTION ERROR"
169 end if
170 if the errorNumber is not -128 then
171 tell application id "com.apple.iWork.Numbers"
172 activate
173 display alert alertString message errorMessage buttons {"Cancel"}
174 error number -128
175 end tell
176 end if
177end try
178 
179on incrementtableNameIfNeeded(thistableName, thisDivider)
180 -- a routine for deriving a unique table name based on a provided table name
181 tell application "Numbers"
182 tell the front document
183 tell the active sheet
184 -- get a list of the names of the tables on the active sheet
185 set the thesetableNames to the name of every table
186 set the testtableName to the thistableName
187 set incrementor to 0
188 -- increment the passed name, if needed
189 repeat until testtableName is not in thesetableNames
190 set incrementor to incrementor + 1
191 set the testtableName to ¬
192 thistableName & thisDivider & (incrementor as string)
193 end repeat
194 -- return the non-conflicting table name
195 return the testtableName
196 end tell
197 end tell
198 end tell
199end incrementtableNameIfNeeded
