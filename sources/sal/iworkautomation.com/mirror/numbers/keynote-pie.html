Exporting to Keynote: Pie Chart

Here’s an example AppleScript script that transforms the data from a single selected row or column in a Numbers table to a pie chart on a new slide in Keynote.

DO THIS ► DOWNLOAD the simple example Numbers document.

Simply select a single row or column in a Numbers table and run the script. The row data will be extracted and used to create a pie chart on a new slide in the frontmost Keynote document. In addition, the name of the selected table can optionally be used as the title for the created slide.

New Pie Chart in Keynote using Data From Selected Column | Row
  
01property useTableNameForslideTitle : false
02 
03tell application "Numbers"
04 activate
05 try
06 -- check for document
07 if not (exists document 1) then error number 1000
08 tell document 1
09 try -- check for selected table
10 tell active sheet
11 set the selectedTable to ¬
12 (the first table whose class of selection range is range)
13 end tell
14 on error
15 error number 1001
16 end try
17 tell selectedTable
18 set thisTableName to its name
19 -- get the count fo the various headers and footers
20 set the headercolumnCount to header column count
21 set the headerrowCount to header row count
22 set the footerrowCount to footer row count
23 -- store references to the intersecting columns and rows
24 set theseRows to the rows of the selection range
25 set theseColumns to the columns of the selection range
26 -- get the row and column counts
27 set rowCount to (count of theseRows)
28 set columnCount to (count of theseColumns)
29 -- check for errors
30 if (rowCount is not 1) and (columnCount is not 1) then
31 error number 1002
32 else if (rowCount is 1) and (columnCount is 1) then
33 error number 1002
34 end if
35 if headerrowCount is greater than 1 then
36 error number 1003
37 end if
38 if headercolumnCount is greater than 1 then
39 error number 1004
40 end if
41 -- get the column names
42 set columnNames to {}
43 if headercolumnCount is not 0 then
44 repeat with i from 1 to the count of theseColumns
45 set thisValue to (the value of the first cell of (item i of theseColumns))
46 if thisValue is not missing value then
47 set the end of columnNames to thisValue
48 end if
49 end repeat
50 end if
51 -- get the row names
52 set rowNames to {}
53 if headerrowCount is not 0 then
54 repeat with i from 1 to the count of theseRows
55 set thisValue to (the value of the first cell of (item i of theseRows))
56 if thisValue is not missing value then
57 set the end of rowNames to thisValue
58 end if
59 end repeat
60 end if
61 end tell
62 -- add the chart to a new slide inserted after the current slide
63 if rowCount is greater than columnCount then
64 -- a column is selected
65 tell selectedTable
66 set thisColumn to item 1 of theseColumns
67 set footerOffset to ((1 + footerrowCount) * -1)
68 set dataSet to ¬
69 (the value of cells (headercolumnCount + 1) thru footerOffset of thisColumn)
70 end tell
71 tell application "Keynote"
72 activate
73 if not (exists document 1) then make new document
74 tell the front document
75 set thisSlide to ¬
76 make new slide at after (get current slide) with properties ¬
77 {base slide:master slide "Title - Top"}
78 tell thisSlide
79 if useTableNameForslideTitle is true then
80 set slideTitle to thisTableName
81 else
82 set slideTitle to (item 1 of the columnNames)
83 end if
84 set the object text of the default title item to slideTitle
85 add chart row names columnNames column names rowNames ¬
86 data {dataSet} type pie_2d group by chart column
87 end tell
88 end tell
89 end tell
90 else
91 -- a row is selected
92 tell selectedTable
93 set thisRow to item 1 of theseRows
94 set footerOffset to ((1 + footerrowCount) * -1)
95 set dataSet to ¬
96 (the value of cells (headerrowCount + 1) thru footerOffset of thisRow)
97 end tell
98 tell application "Keynote"
99 activate
100 if not (exists document 1) then make new document
101 tell the front document
102 set thisSlide to ¬
103 make new slide at after (get current slide) with properties ¬
104 {base slide:master slide "Title - Top"}
105 tell thisSlide
106 if useTableNameForslideTitle is true then
107 set slideTitle to thisTableName
108 else
109 set slideTitle to (item 1 of the rowNames)
110 end if
111 set the object text of the default title item to slideTitle
112 add chart row names rowNames column names columnNames ¬
113 data {dataSet} type pie_2d group by chart column
114 end tell
115 end tell
116 end tell
117 end if
118 end tell
119 on error errorMessage number errorNumber
120 if errorNumber is 1000 then
121 set alertString to "MISSING RESOURCE"
122 set errorMessage to ¬
123 "Please create or open a document before running this script."
124 else if errorNumber is 1001 then
125 set alertString to "SELECTION ERROR"
126 set errorMessage to "Please select a table before running this script."
127 else if errorNumber is 1002 then
128 set alertString to "SELECTION ERROR"
129 set errorMessage to "Please select a single column or row."
130 else if errorNumber is 1003 then
131 set alertString to "TOO MANY ROW HEADERS"
132 set errorMessage to ¬
133 "This script is designed to work with no more than a single row header."
134 else if errorNumber is 1004 then
135 set alertString to "TOO MANY COLUMN HEADERS"
136 set errorMessage to ¬
137 "This script is designed to work with no more than a single column header."
138 else
139 set alertString to "EXECUTION ERROR"
140 end if
141 if errorNumber is not -128 then
142 display alert alertString message errorMessage buttons {"Cancel"}
143 end if
144 error number -128
145 end try
146end tell
