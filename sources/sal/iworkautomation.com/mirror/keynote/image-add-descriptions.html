Add Image Descriptions

Here are two scripts for adding descriptions over or below images in a Keynote presentation.

Add Descriptions Under Images

This script positions image descriptions below the source image frame. If an image has no description, no text container is added. There are two options for controlling the process:

  • You can choose to process all images on all slides, or just the images on the current slide.
  • The font size for the descriptions can be a fixed size or be adjusted based upon the ratio of text length and image width.

Short descriptions:

Long descriptions:

Set the value for the default font family and size properties to suite your requirements.

Add Descriptions Under Images
  
01property offsetPadding : 8
02property textTypeface : "Helvetica Neue Medium"
03property textFontSizeFixed : 24
04property textFontSizeLarge : 36
05property textFontSizeMedium : 24
06property textFontSizeSmall : 18
07property textRGBColorValues : {65535, 65535, 65535}
08 
09tell application "Keynote"
10 activate
11 if playing is true then tell the front document to stop
12 
13 if not (exists document 1) then error number -128
14 
15 tell the front document
16 set documentWidth to its width
17 set documentHeight to its height
18 
19 display dialog "Add descriptions below images on all slides, or just the current slide?" buttons {"Cancel", "All Slides", "Current Slide"} default button 3 with icon 1
20 set targetScope to button returned of the result
21 log ("targetScope: " & targetScope)
22 if the targetScope is "All Slides" then
23 set theseSlides to every slide whose skipped is false
24 else
25 set theseSlides to {}
26 set the end of theseSlides to the current slide
27 end if
28 
29 display dialog "Use fixed or adjustable size for the font size?" buttons {"Cancel", "Adjustable", "Fixed"} default button 3 with icon 1
30 set fontSizeMethod to the button returned of the result
31 log ("fontSizeMethod: " & fontSizeMethod)
32 
33 repeat with i from 1 to the count of theseSlides
34 set thisSlide to item i of theseSlides
35 set imageCount to the count of images of thisSlide
36 repeat with q from 1 to the imageCount
37 set thisImage to image q of thisSlide
38 set thisDescription to the description of thisImage
39 if thisDescription is not "" then
40 copy (my deriveBoundsOfVisibleImageMaskFor(thisImage)) to {x, y, x1, y1}
41 tell thisSlide
42 set thisTextItem to make new text item with properties {position:{x, y}, width:(x1 - x), object text:thisDescription}
43 end tell
44 tell thisTextItem
45 tell object text
46 set font to textTypeface
47 if fontSizeMethod is "Adjustable" then
48 set fontSizeFactor to (x1 - x) div (length of thisDescription)
49 log "fontSizeFactor: " & fontSizeFactor
50 if fontSizeFactor is less than 25 then
51 set textFontSize to textFontSizeSmall
52 else if fontSizeFactor is less than 35 then
53 set textFontSize to textFontSizeMedium
54 else
55 set textFontSize to textFontSizeLarge
56 end if
57 else
58 set textFontSize to textFontSizeFixed
59 end if
60 set size to textFontSize
61 -- Always use “of it” when setting the value of the color property
62 set color of it to textRGBColorValues
63 end tell
64 -- position the text item below the host image
65 set textItemHeight to its height
66 copy its position to {tx, ty}
67 set its position to {tx, (y1 + offsetPadding)}
68 end tell
69 end if
70 end repeat
71 end repeat
72 end tell
73end tell
74 
75on deriveBoundsOfVisibleImageMaskFor(thisImage)
76 tell application "Keynote"
77 tell the front document
78 set documentWidth to its width
79 set documentHeight to its height
80 
81 copy position of thisImage to {imageMaskTopLeftHorizontalOffset, imageMaskTopLeftVerticalOffset}
82 set imageMaskWidth to width of thisImage
83 set imageMaskHeight to height of thisImage
84 
85 -- derive the top left horizontal offset of the visible image mask
86 if imageMaskTopLeftHorizontalOffset is less than 0 then
87 set visbleImageMaskTopLeftHorizontalOffset to 0
88 else
89 set visbleImageMaskTopLeftHorizontalOffset to imageMaskTopLeftHorizontalOffset
90 end if
91 
92 -- derive the bottom right horizontal offset of the visible image mask
93 set imageMaskBottomRightHorizontalOffset to imageMaskTopLeftHorizontalOffset + imageMaskWidth
94 if imageMaskBottomRightHorizontalOffset is greater than documentWidth then
95 set visbleImageMaskBottomRightHorizontalOffset to documentWidth
96 else
97 set visbleImageMaskBottomRightHorizontalOffset to imageMaskBottomRightHorizontalOffset
98 end if
99 
100 -- derive the width of the visible image mask
101 set visbleImageMaskWidth to visbleImageMaskBottomRightHorizontalOffset - visbleImageMaskTopLeftHorizontalOffset
102 
103 log "visbleImageMaskWidth: " & visbleImageMaskWidth
104 
105 -- derive the top left vertical offset of the visible image mask
106 if imageMaskTopLeftVerticalOffset is less than 0 then
107 set visbleImageMaskTopLeftVerticalOffset to 0
108 else
109 set visbleImageMaskTopLeftVerticalOffset to imageMaskTopLeftVerticalOffset
110 end if
111 
112 log "visbleImageMaskTopLeftVerticalOffset: " & visbleImageMaskTopLeftVerticalOffset
113 
114 -- derive the bottom right vertical offset of the visible image mask
115 set imageMaskBottomRightVerticalOffset to imageMaskTopLeftVerticalOffset + imageMaskHeight
116 if imageMaskBottomRightVerticalOffset is greater than documentHeight then
117 set visibleImageMaskBottomRightVerticalOffset to documentHeight
118 else
119 set visibleImageMaskBottomRightVerticalOffset to imageMaskBottomRightVerticalOffset
120 end if
121 
122 log "visibleImageMaskBottomRightVerticalOffset: " & visibleImageMaskBottomRightVerticalOffset
123 
124 -- derive the height of the visible image mask
125 set visbleImageMaskHeight to visibleImageMaskBottomRightVerticalOffset - visbleImageMaskTopLeftVerticalOffset
126 
127 log "visbleImageMaskheight: " & visbleImageMaskHeight
128 
129 -- return bounds of the visible image mask
130 return {visbleImageMaskTopLeftHorizontalOffset, visbleImageMaskTopLeftVerticalOffset, visbleImageMaskBottomRightHorizontalOffset, visibleImageMaskBottomRightVerticalOffset}
131 end tell
132 end tell
133end deriveBoundsOfVisibleImageMaskFor

Overlay Image Descriptions

Rather than place the image description below the source image, the following script overlays the description over the bottom of the source image.

Overlay Image Descriptions
  
01property overlayPadding : 18
02property textOverlayTypeface : "Helvetica"
03property textFontSizeFixed : 28
04property textOverlayFontSizeLarge : 36
05property textOverlayFontSizeMedium : 24
06property textOverlayFontSizeSmall : 18
07property textOverlayRGBColorValues : {65535, 65535, 65535}
08 
09tell application "Keynote"
10 activate
11 if not (exists document 1) then error number -128
12 if playing is true then stop every document
13 display dialog "Overlay the description or file name on every image in this presentation?" buttons {"Cancel", "File Name", "Description"} default button 3 with icon 1
14 set chosenMetadata to the button returned of the result
15 log ("chosenMetadata: " & chosenMetadata)
16 display dialog "Overlay descriptions on all slides, or just the current slide?" buttons {"Cancel", "All Slides", "Current Slide"} default button 3 with icon 1
17 set targetScope to button returned of the result
18 log ("targetScope: " & targetScope)
19 if the targetScope is "All Slides" then
20 set theseSlides to every slide whose skipped is false
21 else
22 set theseSlides to {}
23 set the end of theseSlides to the current slide of document 1
24 end if
25 display dialog "Use fixed or adjustable size for the font size?" buttons {"Cancel", "Adjustable", "Fixed"} default button 3 with icon 1
26 set fontSizeMethod to the button returned of the result
27 log ("fontSizeMethod: " & fontSizeMethod)
28 tell the front document
29 set documentWidth to its width
30 set documentHeight to its height
31 repeat with i from 1 to the count of theseSlides
32 set thisSlide to item i of theseSlides
33 set imageCount to the count of images of thisSlide
34 repeat with q from 1 to the imageCount
35 set thisImage to image q of thisSlide
36 if chosenMetadata is "Description" then
37 set thisDescription to the description of thisImage
38 else
39 set thisDescription to the file name of thisImage
40 end if
41 if thisDescription is not "" then
42 copy (my deriveBoundsOfVisibleImageMaskFor(thisImage)) to {x, y, x1, y1}
43 tell thisSlide
44 set thisTextItem to make new text item with properties {position:{x, y}, width:(x1 - x), object text:thisDescription}
45 end tell
46 tell thisTextItem
47 tell object text
48 set font to textOverlayTypeface
49 if fontSizeMethod is "Adjustable" then
50 set fontSizeFactor to (x1 - x) div (length of thisDescription)
51 log "fontSizeFactor: " & fontSizeFactor
52 if fontSizeFactor is less than 20 then
53 set textOverlayFontSize to textOverlayFontSizeSmall
54 else if fontSizeFactor is less than 25 then
55 set textOverlayFontSize to textOverlayFontSizeMedium
56 else
57 set textOverlayFontSize to textOverlayFontSizeLarge
58 end if
59 else
60 set textOverlayFontSize to textFontSizeFixed
61 end if
62 set size to textOverlayFontSize
63 set color of it to textOverlayRGBColorValues
64 end tell
65 -- position the text overlay at bottom of host image
66 set textItemHeight to its height
67 copy its position to {tx, ty}
68 set its position to {tx, (y1 - textItemHeight - overlayPadding)}
69 end tell
70 end if
71 end repeat
72 end repeat
73 end tell
74end tell
