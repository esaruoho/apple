Generate QR Code Image

This handy script uses AppleScriptObjective-C to create a QR code image based upon the text contained on the clipboard, and imports the created image onto the current slide. It’s a very useful tool for those that often display URLs in their presentations.

NOTE: The following script is written using AppleScriptObj-C, an incredibly powerful fusion of the AppleScript and Objective-C languages. To acquire detailed training materials and examples about AppleScriptObj-C visit: macosxautomation.com/applescript/apps

To use this script, simply copy the URL or text to convert into a QR code to the clipboard, and then run the script. A QR code image will be created and added to the current slide.

Add QR Code using Clipboard Text to Current Slide
  
01use framework "Foundation"
02use framework "CoreImage"
03use framework "AppKit"
04use framework "CoreGraphics"
05use scripting additions
  
06property targetMinimumDimension : 400
  
07try
08 tell application "Keynote"
09 activate
10 if playing is true then tell the front document to stop
11 if not (exists document 1) then error number -128
12 if (clipboard info for «class utf8») is not {} then
13 set clipboardText to (get the clipboard as «class utf8»)
14 else
15 error "There is no text on the clipboard to convert into a QR code image."
16 end if
17 end tell
18 
19 -- derive a file path for the created image file
20 set theUUID to current application's NSUUID's UUID()'s UUIDString()
21 set fileManager to current application's NSFileManager's defaultManager()
22 set theURL to (fileManager's URLsForDirectory:(current application's NSPicturesDirectory) inDomains:(current application's NSUserDomainMask))'s firstObject()
23 set the targetItemHFSpath to (theURL as string) & (theUUID as string) & ".jpg"
24 set targetItemPOSIXpath to POSIX path of targetItemHFSpath
25 
26 set thisImageObject to my createFullSizeQRCodeImageObjectForString(clipboardText)
27 my writeNSImageObjectToFileAsJPEG(thisImageObject, targetItemPOSIXpath, false)
28 
29 set the targetFile to targetItemHFSpath as alias
30 tell application "Keynote"
31 activate
32 if not (exists document 1) then
33 make new document
34 end if
35 tell the front document
36 set documentWidth to the width of it
37 set documentHeight to the height of it
38 tell current slide
39 -- add the image to the slide and size to desired dimension
40 set thisImage to make new image with properties {file:targetFile, width:targetMinimumDimension, height:targetMinimumDimension}
41 -- center the added image on the slide
42 set the position of thisImage to {(documentWidth - targetMinimumDimension) div 2, (documentHeight - targetMinimumDimension) div 2}
43 end tell
44 end tell
45 end tell
46 
47 -- Since the image is automatically added to document bundle, move the source image file to trash
48 set fileManager to current application's NSFileManager's defaultManager()
49 set thisPOSIXpath to current application's NSString's stringWithString:targetItemPOSIXpath
50 set resultingURL to missing value
51 set URLOfItemToTrash to (current application's |NSURL|'s fileURLWithPath:thisPOSIXpath)
52 (fileManager's trashItemAtURL:URLOfItemToTrash resultingItemURL:resultingURL |error|:(missing value))
53 
54 say "Done!"
55on error errorMessage number errorNumber
56 if errorNumber is not -128 then
57 set spokenErrorTitle to "I was unable to complete your request: "
58 set cfgutil to "/usr/bin/say"
59 set theTask to (current application's NSTask's launchedTaskWithLaunchPath:cfgutil arguments:{(spokenErrorTitle & errorMessage)})
60 tell application (POSIX path of (path to frontmost application))
61 activate
62 display alert "Unable to Complete Request:" message errorMessage buttons {"Cancel"} default button 1
63 end tell
64 theTask's terminate()
65 end if
66end try
67 
68on createFullSizeQRCodeImageObjectForString(thisString)
69 -- returns a full-size NSImage object
70 
71 set thisString to current application's NSString's stringWithString:thisString
72 
73 set thisData to thisString's dataUsingEncoding:(current application's NSUTF8StringEncoding)
74 
75 set anImageFilter to current application's CIFilter's filterWithname:"CIQRCodeGenerator"
76 anImageFilter's setDefaults()
77 anImageFilter's setValue:thisData forKey:"inputMessage"
78 anImageFilter's setValue:"L" forKey:"inputCorrectionLevel"
79 
80 set baseImage to anImageFilter's outputImage()
81 
82 set aTransform to current application's CGAffineTransform's CGAffineTransformMakeScale(100.0, 100.0)
83 
84 set outputImage to baseImage's imageByApplyingTransform:aTransform
85 
86 set imageRepresentation to current application's NSCIimageRep's imageRepWithCIImage:outputImage
87 
88 set resultingImageObject to current application's NSImage's alloc()'s initWithSize:(imageRepresentation's |size|())
89 
90 resultingImageObject's addRepresentation:imageRepresentation
91 
92 return resultingImageObject
93end createFullSizeQRCodeImageObjectForString
94 
95on createScaledQRCodeImageObjectForString(thisString, targetWidth, targetHeight)
96 -- returns a scaled NSImage object
97 
98 set thisString to current application's NSString's stringWithString:thisString
99 
100 set thisData to thisString's dataUsingEncoding:(current application's NSUTF8StringEncoding)
101 
102 set anImageFilter to current application's CIFilter's filterWithname:"CIQRCodeGenerator"
103 anImageFilter's setDefaults()
104 anImageFilter's setValue:thisData forKey:"inputMessage"
105 anImageFilter's setValue:"L" forKey:"inputCorrectionLevel"
106 
107 set baseImage to anImageFilter's outputImage()
108 
109 set baseImageSize to baseImage's extent()'s |size|()
110 
111 set xScale to targetWidth / (baseImageSize's width())
112 set yScale to targetHeight / (baseImageSize's height())
113 
114 set aTransform to current application's CGAffineTransform's CGAffineTransformMakeScale(xScale, yScale)
115 
116 set outputImage to baseImage's imageByApplyingTransform:aTransform
117 
118 set imageRepresentation to current application's NSCIimageRep's imageRepWithCIImage:outputImage
119 
120 set resultingImageObject to current application's NSImage's alloc()'s initWithSize:(imageRepresentation's |size|())
121 
122 resultingImageObject's addRepresentation:imageRepresentation
123 
124 return resultingImageObject
125end createScaledQRCodeImageObjectForString

127on writeNSImageObjectToFileAsJPEG(thisImageObject, targetImageFilePath, shouldRevealInFinder)
128 -- create JPEG data for the image object
129 set tiffData to thisImageObject's TIFFRepresentation()
130 set imageRep to current application's NSBitmapimageRep's imageRepWithData:tiffData
131 set theProps to current application's NSDictionary's dictionaryWithObject:1.0 forKey:(current application's NSImageCompressionFactor)
132 set imageData to (imageRep's representationUsingType:(current application's NSJPEGFileType) |properties|:theProps)
133 
134 -- write the JPEG data to file
135 set theResult to (imageData's writeTofile:targetImageFilePath atomically:true |error|:(missing value)) as boolean
136 if theResult is true then
137 if shouldRevealInFinder is true then
138 set theseURLs to {}
139 set the end of theseURLs to (current application's NSURL's fileURLWithPath:targetImageFilePath)
140 -- reveal items in file viewer
141 tell current application's NSWorkspace to set theWorkspace to sharedWorkspace()
142 tell theWorkspace to activateFileViewerSelectingURLs:theseURLs
143 end if
144 return true
145 else
146 error "There was a problem writing the image object to file."
147 end if
148end writeNSImageObjectToFileAsJPEG
